```python
# app/core/uptc_router_integration.py

class UPTCRouterIntegration:
    """Integration between Guard Orchestrator (Orbit 3) and UPTC Router (Orbit 4)."""
    
    def __init__(self, uptc_core: UPTCCore):
        self.uptc_core = uptc_core
        self.adapter = OrchestrationAdapter()
    
    async def route_request(
        self,
        request: OrchestrationRequest
    ) -> Optional[str]:
        """Route OrchestrationRequest via UPTC Router."""
        # Transform to UPTCMessage
        uptc_msg = self.adapter.orchestration_request_to_uptc_message(request)
        
        # Route via UPTC Unified Router
        if self.uptc_core.unified_router:
            target_agent = self.uptc_core.unified_router.route(uptc_msg)
            if target_agent:
                return target_agent
        
        # Fallback to explicit routing
        return request.service_type.value
```

```python
        # Transform to UPTCMessage
        uptc_msg = self.adapter.orchestration_request_to_uptc_message(request)
        
        # Route via UPTC Unified Router
        if self.uptc_core.unified_router:
            target_agent = self.uptc_core.unified_router.route(uptc_msg)
            if target_agent:
                return target_agent
        
        # Fallback to explicit routing
        return request.service_type.value
```
