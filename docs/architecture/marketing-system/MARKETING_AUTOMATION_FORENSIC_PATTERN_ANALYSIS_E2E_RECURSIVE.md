def execute_with_retry(
    operation: callable,
    max_retries: int = 3,
    backoff: float = 1.0
) -> Tuple[Any, bool]:
    """
    Recursive error recovery pattern
    
    Applied to:
    - API failures (Google Ads, LinkedIn, Facebook, Instagram)
    - Email delivery failures
    - Deployment failures
    - Credential loading failures
    - Validation failures
    """
    for attempt in range(max_retries):
        try:
            result = operation()
            return result, True
        except Exception as e:
            if attempt < max_retries - 1:

            else:
                # Final failure
                return None, False
